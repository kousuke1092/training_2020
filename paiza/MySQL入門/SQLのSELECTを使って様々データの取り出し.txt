■SELECT命令を使って様々な情報を取り出す

-- データ件数を表示する
SELECT COUNT(*) FROM テーブル名;
	//COUNT(*)で件数カウント
	処理内容=>テーブル名のデータをカウントして出力

--条件に合ったデータの件数を表示する
SELECT COUNT(*) FROM テーブル名 WHERE カラム名 = 数値;
	//WHEREで条件を指定してカウントすることも可能
	処理内容=>テーブル名のカラム名の中で数値のものをカウントして出力

-- データを並び替えて取得する
SELECT * FROM テーブル名 ORDER BY カラム名;
	//ORDER BYで少ない値から並び替えて取り出すことが可能
	処理内容=>テーブルの全てのカラムからカラム名を並び替えて取り出し
-- データを並び替えて取得する 逆順
SELECT * FROM players ORDER BY level DESC;
	//文末にDESCをつけることで逆順(大きい順)にすることも可能

--上位3件だけ表示す
SELECT * FROM テーブル名 ORDER BY カラム名 DESC LIMIT 数値;
	//ORDER BY カラム名 DESCで逆順並び替え、LIMITで上位何件かだけ取ることが可能
	処理内容=>テーブル名のすべてのカラムからカラム名を逆順並び替えし、上位 数値 位まで取り出し。

-- カラム名ごとに人数を集計する
SELECT カラム名, COUNT(*) FROM テーブル名 GROUP BY カラム名;
	//GROUP BYでグループごとに集計して表示することが可能
	処理内容=>テーブル名のカラム名からカラム名の数を集計し取り出す




英語を訳すように考えれば意外と処理がわかる。
