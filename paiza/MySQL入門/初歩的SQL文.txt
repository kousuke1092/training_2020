-- 全てのデータを取り出す
SELECT * FROM テーブル名;
   //先頭は命令、アスタリスクはすべてのカラムという意味
   //SELECTで取り出す命令、FROM テーブル名で対象指定。
       処理内容=>テーブル名からすべてのカラムを取り出す

-- 一部のカラムだけ取得する
SELECT カラム名1,  カラム名2 テーブル名;
   //カンマ区切りで複数取り出し可能。
   　　処理内容=>テーブル名からカラム名１とカラム名２を取り出す

-- 一部の行だけ取得する
SELECT * FROM テーブル名 WHERE カラム名 >= 行数;
   //WHEREで特定の条件指定ができる
   　　処理内容=>テーブル名のすべてのデータからカラム名の行数を取り出す

WHEREの条件指定には
　=   等しい
  <　 小さい
  >　 大きい
  <=  以下
  >=  以上
  <>　等しくない

  AND 両方の条件が成立した場合
  OR　どちらか一方が成立した場合

が使える。

-- 複数の条件を組み合わせる
SELECT * FROM テーブル名 WHERE カラム名１ >= 数値１ AND カラム名２ <> 数値２;
　　//ANDを使うと複数条件指定ができる。
　　　　処理内容=>テーブル名のすべてのカラムからカラム名1が数値1以上と
　　　　　　　　　カラム名2が数値２と等しくないデータだけを取り出す

--条件指定とカラム選択を組み合わせる
SELECT カラム名1, カラム名2 FROM テーブル名 WHERE カラム名1 >= 数値;
　　//このように絞り込みをして取り出すこともできる
　　　　処理内容=>テーブル名のカラム名1と2からカラム名1が数値以上のデータを取り出す。
